实现多次执行脚本后，只有一个脚本进程

#!/bin/sh
#Author: dengpangpang
#Date: 2024-01-09
#Description: 实现执行多个相同脚本后，只有一个脚本进程

#定义一个脚本进程的pid文件
pidpath=/tmp/a.pid
#如果已经存在了pid文件，则杀掉已经存在的脚本进程
if [ -e "${pidpath}" ]
  then
    kill $(cat ${pidpath}) > /dev/null 2>&1
    rm -f ${pidpath}
fi
#重新写入，将当前脚本进程的pid号写入pid文件
echo $$ > ${pidpath}
sleep 300
