  2. 修改任务文件

    # 修改history_work.collector.yml任务文件，修改相关数据库，demo替换为客户简拼
vim /usr/local/sql_exporter_demo/history_work.collector.yml

collector_name: history_work
metrics:
  - metric_name: history_wrok
    type: counter
    help: '监控sqlserver历史作业执行状态:0表示失败,1表示成功,2表示重新尝试,3表示取消,4表示进行中'
    key_labels:
      - workname
    value_label: operation
    values:
      - workstatus
      - workdata
      - worktime
      - spendtime
    query: |
      SELECT jobs.name AS workname,history.run_date AS workdata,history.run_time AS worktime,history.run_status AS workstatus,history.run_duration AS spendtime
      FROM msdb.dbo.sysjobhistory AS history
      LEFT JOIN msdb.dbo.sysjobs as jobs ON history.job_id = jobs.job_id
      WHERE history.run_date = convert(varchar(100),getdate(),112)  AND history.step_name LIKE '%(作业结果)%' 

  - metric_name: history_wrok_row
    type: gauge
    help: '统计客户历史作业执行情况'
    values: [history_wrok_row]
    query: |
      SELECT count(*) history_wrok_row FROM msdb.dbo.sysjobhistory AS history WHERE  history.run_date = convert(varchar(100),getdate(),112) AND history.run_status != 1
#备注：以下修改数据库名称.Inroad-DEMO替换为客户数据库名称
  - metric_name: job_ticket_number
    type: gauge
    help: '客户当日开票的数量'
    values: [job_ticket]
    query: |
      SELECT COUNT(*) job_ticket FROM [Inroad-DEMO].dbo.SafeOperationLicenseRecord AS a WHERE datediff(dd, a.planbegintime,getdate())=0 AND isdeleted=0

  - metric_name: effective_person_number
    type: gauge
    help: '统计客户有效人员的数量'
    values: [effective_person]
    query: |
      SELECT count(c_id) AS effective_person FROM [Inroad-DEMO].dbo.Accounts WHERE status = 1

  - metric_name: inspection_number
    type: gauge
    help: '统计客户当天巡检数量'
    values: [inspection]
    query: |
      select count(1) AS inspection from [Inroad-DEMO].dbo.InspectionPlanRecord  WHERE c_createtime > convert(nvarchar(10),getdate(),121)

  - metric_name: hiddanger_number
    type: gauge
    help: '统计客户当天隐患的数量'
    values: [hiddanger]
    query: |
      select count(1) AS hiddanger from [Inroad-DEMO].dbo.PotentialDangerRecord WHERE c_createtime > convert(nvarchar(10),getdate(),121)

  3. 修改配置文件

    # 修改配置文件，demo替换为客户简拼

global:
  scrape_timeout_offset: 500ms
  min_interval: 0s
  max_connections: 3
  max_idle_connections: 3
target:
  # Data source name always has a URI schema that matches the driver name. In some cases (e.g. MySQL)
  # the schema gets dropped or replaced to match the driver expected DSN format.
  data_source_name: 'sqlserver://数据库用户名:数据库密码@数据库地址:数据库端口'  #内网默认是1433,外网是在比格云控制台设置的转发端口
  密码如果有特殊字符需要转义

  collectors: [mssql_standard, history_work] 
  collector_files: 
  - "*.collector.yml"
 
#重新加载服务启动项
systemctl daemon-reload
# 启动服务
systemctl start sql_demo.service
#设置开机自启
systemctl enable  sql_demo.service
