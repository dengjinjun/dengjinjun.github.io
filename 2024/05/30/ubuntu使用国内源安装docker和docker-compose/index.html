阿里云方案

#!/bin/bash
# author: dengjinjun
# date: 2024-05-30
# description: auto install docker docker-compose

echo "============================================开始执行脚本,卸载旧版本docker==============================================="
echo " "
echo " "
echo " "
echo " "
apt-get remove docker docker-engine docker.io containerd runc 
if [ $? -eq 0 ]; then
    echo "=======================================卸载旧版本的docker完成，开始下载工具和依赖项=========================================="
else
    echo "卸载失败" && exit 1
fi

apt-get update
apt-get install -y apt-transport-https ca-certificates curl gnupg lsb-release docker-compose-plugin
if [ $? -eq 0 ]; then
    echo "=======================================工具和依赖项下载完成，开始下载并添加docker密钥========================================"
else
    echo "安装工具和依赖项失败" && exit 1
fi

curl -fsSL https://mirrors.aliyun.com/docker-ce/linux/ubuntu/gpg |  gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg

if [ $? -eq 0 ]; then
    echo "==========================================docker密钥添加完成，开始添加docker源============================================"
else
    echo "docker密钥添加失败" && exit 1
fi
echo "deb [arch=amd64 signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://mirrors.aliyun.com/docker-ce/linux/ubuntu $(lsb_release -cs) stable" | tee /etc/apt/sources.list.d/docker.list > /dev/null

if [ $? -eq 0 ]; then
    echo "============================================docker源添加完成，开始安装docker============================================="
else
    echo "docker源添加失败" && exit 1
fi
apt-get update
apt-get install -y docker-ce docker-ce-cli containerd.io docker-compose-plugin
systemctl enable docker
if [ $? -eq 0 ]; then
    echo "======================================安装完成，已设置自启动，开始安装docker-compose========================================"
else
    echo "docker安装失败" && exit 1
fi

curl -L "https://3dview-1251252938.cos.ap-shanghai.myqcloud.com/dengjinjun/software/docker-compose" -o /usr/local/bin/docker-compose && \
chmod +x /usr/local/bin/docker-compose  && \
ln -s /usr/local/bin/docker-compose /usr/bin/docker-compose
if [ $? -eq 0 ]; then
    echo "===============================================docker-compose安装完成==============================================="
else
    echo "docker-compose安装失败" && exit 1
fi

docker login -u road-1661826238505 -p 4678a3ca6f3ba64013eed03b6e9e94be0f639fa9 in-road-docker.pkg.coding.net
echo "===============================================镜像仓库登录成功，开始拉取镜像==============================================="
docker-compose pull
echo "===============================================镜像拉取完成，等待手动启动容器==============================================="

清华源方案（备选）

#!/bin/bash
# author: dengjinjun
# date: 2024-05-30
# description: auto install docker docker-compose


# 写入清华apt源
cp  -f /etc/apt/sources.list /etc/apt/sources.list.backup
cat > /etc/apt/sources.list <<EOF
deb https://mirrors.tuna.tsinghua.edu.cn/ubuntu/ bionic main restricted universe multiverse
deb-src https://mirrors.tuna.tsinghua.edu.cn/ubuntu/ bionic main restricted universe multiverse
deb https://mirrors.tuna.tsinghua.edu.cn/ubuntu/ bionic-updates main restricted universe multiverse
deb-src https://mirrors.tuna.tsinghua.edu.cn/ubuntu/ bionic-updates main restricted universe multiverse
deb https://mirrors.tuna.tsinghua.edu.cn/ubuntu/ bionic-backports main restricted universe multiverse
deb-src https://mirrors.tuna.tsinghua.edu.cn/ubuntu/ bionic-backports main restricted universe multiverse
deb https://mirrors.tuna.tsinghua.edu.cn/ubuntu/ bionic-security main restricted universe multiverse
deb-src https://mirrors.tuna.tsinghua.edu.cn/ubuntu/ bionic-security main restricted universe multiverse
deb https://mirrors.tuna.tsinghua.edu.cn/ubuntu/ bionic-proposed main restricted universe multiverse
deb-src https://mirrors.tuna.tsinghua.edu.cn/ubuntu/ bionic-proposed main restricted universe multiverse
EOF
if [ $? -eq 0 ]; then
    echo "===================================================写入清华apt源完成==================================================="
else
    echo "清华apt源写入失败" && exit 1
fi

# 更新软件包
apt update -y

# 下载密钥
{ curl -fsSL https://download.docker.com/linux/ubuntu/gpg | apt-key add -; }
if [ $? -eq 0 ]; then
    echo "===================================================密钥下载完成==================================================="
else
    echo "密钥下载失败" && exit 1
fi

# 添加源
add-apt-repository "deb [arch=amd64] https://mirrors.tuna.tsinghua.edu.cn/docker-ce/linux/ubuntu $(lsb_release -cs) stable"
if [ $? -eq 0 ]; then
    echo "===================================================apt源添加完成==================================================="
else
    echo "apt源添加失败" && exit 1
fi
# 再次更新软件包
apt update -y


# 安装docker和docker-compose
apt -y install docker-ce docker-ce-cli containerd.io docker-compose-plugin
if [ $? -eq 0 ]; then
    echo "===================================================docker安装完成==================================================="
else
    echo "docker安装失败" && exit 1
fi
systemctl enable docker

# 安装docker-compose
curl -L "https://3dview-1251252938.cos.ap-shanghai.myqcloud.com/dengjinjun/software/docker-compose" -o /usr/local/bin/docker-compose && \
chmod +x /usr/local/bin/docker-compose  && \
ln -s /usr/local/bin/docker-compose /usr/bin/docker-compose
if [ $? -eq 0 ]; then
    echo "===============================================docker-compose安装完成==============================================="
else
    echo "docker-compose安装失败" && exit 1
fi

docker login -u road-1661826238505 -p 4678a3ca6f3ba64013eed03b6e9e94be0f639fa9 in-road-docker.pkg.coding.net
echo "===============================================镜像仓库登录成功，开始拉取镜像==============================================="
docker-compose pull
echo "===============================================镜像拉取完成，等待手动启动容器==============================================="
