手动部分

配置ES索引

  1. 创建生命周期策略

    ESweb页面侧边栏，点击开发工具，输入语句后执行：

    demo需要替换为客户小写简拼，code需要替换为客户代码

    PUT _ilm/policy/demo_code_production_policy
{
  "policy": {
    "phases": {
      "hot": {
        "min_age": "0ms",
        "actions": {
          "set_priority": {
            "priority": 100
          }
        }
      },
      "delete": {
        "min_age": "7d",
        "actions": {
          "delete": {
            "delete_searchable_snapshot": true
          }
        }
      }
    }
  }
}

  2. 创建模板

    demo需要替换为客户小写简拼，code需要替换为客户代码

    PUT _template/demo_code_production_template
{
  "index_patterns": ["demo_code_production*"]
}

  3. 生命周期策略匹配模板

    侧边栏——stackmanagement——索引生命周期策略——搜索刚创建的生命周期策略——点击右侧+号，将策略添加到索引模板——显示旧版索引模板——搜索刚创建的索引模板——添加策略

自动部分（放在jenkins中）

#!/bin/bash
# author: dengjinjun
# date: 2024-06-06
# description: add windows and linux info

# ==================================================远程到监控主机上执行=====================================================
CUSTOM_NAME = 
CUSTOM_SPELL = 
DOMAIN = 
CODE = 
WEB_ADDR = 
SQL_ADRR = 
SQL_PASSWD = 

# 添加windows主机信息
cat > /usr/local/prometheus/windows/${CUSTOM_SPELL}.json <<EOF
[
  {
    "targets": [
      "${SQL_ADRR}:9182"
    ],
    "labels": {
      "server": "${CUSTOM_SPELL}-db",
      "user": "${CUSTOM_NAME}"
    }
  }
]
EOF

# 添加linux主机信息
cat > /usr/local/prometheus/node/${CUSTOM_SPELL}.json <<EOF
[
  {
    "targets": [
     "${WEB_ADRR}:9100"
    ],
    "labels": {
     "services": "${CUSTOM_SPELL}-web",
     "user": "${CUSTOM_NAME}"
    }
  }
]
EOF

# 添加数据库监控信息,注意这里的port，要想办法从数据库监控主机上获取到它的值
cat >> /usr/local/prometheus/sqlserver/sqlserver.json <<EOF
,
{
    "targets": [
      "10.100.14.151:$port"
    ],
    "labels": {
      "server": "${CUSTOM_SPELL}",
      "user": "${code}_${CUSTOM_NAME}"
    }
}
EOF

# 重新加载配置文件
curl -X POST http://localhost:9011/-/reload

# ===================================================远程到web主机上执行=====================================================
cd 
wget https://3dview-1251252938.cos.ap-shanghai.myqcloud.com/dengjinjun/scripts/install_exporter.sh
chmod +x install_exporter.sh
sh install_exporter.sh

# ===================================================远程到数据库监控主机上执行=====================================================
cd

declare -i port=9240
find_port() {
    while true; do
        if [ ! -z "$(ss -tunlp | grep $port)" ]; then
            ((port++))
        else
            return "$port"
        fi
    done
}

find_port
sh /root/install_sqlexporter.sh ${CUSTOM_SPELL} ${port}
sed -i "s/DMN/${CUSTOM_SPELL}/g" /usr/local/sql_exporter_${CUSTOM_SPELL}/history_work.collector.yml
sed -i "s/prom_user:prom_password/sa:${SQL_PASSWD}/g" /usr/local/sql_exporter_${CUSTOM_SPELL}/sql_exporter.yml    
sed -i "s/dbserver1.example.com/${SQL_ADDR}/g" /usr/local/sql_exporter_${CUSTOM_SPELL}/sql_exporter.yml
sed -i "s/mssql_standard/mssql_standard,history_work/g" /usr/local/sql_exporter_${CUSTOM_SPELL}/sql_exporter.yml



systemctl daemon-reload
systemctl start sql_${CUSTOM_SPELL}.service
systemctl enable sql_${CUSTOM_SPELL}.service

# ===================================================远程到无锡备份主机上执行=====================================================
cd /data/ && mkdir Inroad-${CUSTOM_SPELL}
chown rsync:rsync Inroad-${CUSTOM_SPELL}

# ===================================================远程到北京备份主机上执行=====================================================
cd /data/ && mkdir Inroad-${CUSTOM_SPELL}
chown rsync:rsync Inroad-${CUSTOM_SPELL}


cat >> /etc/rsyncd.conf <<EOF

[Inroad-DEMO]
path = /data/Inroad-${CUSTOM_SPELL}
EOF

systemctl restart rsyncd
